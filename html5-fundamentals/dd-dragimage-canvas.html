var cancel = function(e) {
  if (e.preventDefault) {
  e.preventDefault();
  }

  if (e.stopPropagation) {
    e.stopPropagation();
  }

  return false;
};

var dragStart = function(e) {
  var 
    canvas = document.createElement('CANVAS'),
    context = canvas.getContext('2d'),
    img = document.getElementById('drag-image'),
    dragImage = null;

  context.drawImage(img, 0, 0);
  dragImage = document.createElement('IMG');
  dragImage.src = canvas.toDataURL();
  e.dataTransfer.setDragImage(dragImage, -15, 15);

  try {
    e.dataTransfer.setData('text/plain', '');
  } catch (e) {
    e.dataTransfer.setData('Text', '');
  }
};


var target = document.getElementById('target');
target.addEventListener('drop', cancel, false);
target.addEventListener('dragenter', cancel, false);
target.addEventListener('dragover', cancel, false);

var source = document.getElementById('source');
source.addEventListener('dragstart', dragStart, false);
